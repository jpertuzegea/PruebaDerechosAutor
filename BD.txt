create database Prueba;
use Prueba;

 CREATE TABLE [LoginLog](
	[Id] [int] IDENTITY(1,1) NOT NULL,
	[UserName] [varchar](30) NULL,	
	[LoginTime] [varchar](30) NULL,	
	[AccessToken]  VARCHAR(MAX)	
	PRIMARY KEY (Id)
	);
